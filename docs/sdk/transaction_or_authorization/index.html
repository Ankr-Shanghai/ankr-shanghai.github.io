    export const getFromatData=(params:any)=>{
    
    // tokenContract：Contract address      abi：abi file
    const contract = new ethers.Contract(params.tokenContract, params.abi);

    // methods：abi method to be called      abiInput：Parameters corresponding to this method
    const data = '0x' + contract.interface.encodeFunctionData(params.methods, params.abiInput).substr(2);

    return data;
}
